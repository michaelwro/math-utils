547inline Vector<3> cross(const Vector<3>& v1, const Vector<3>& v2)
548{
549 return Vector<3>{
550 (v1(1) * v2(2)) - (v1(2) * v2(1)),
551 (v1(2) * v2(0)) - (v1(0) * v2(2)),

553 };
554}
