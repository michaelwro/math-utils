76void Quaternion::normalize()
77{
78 // compute magnitude
79 const double magn = std::sqrt(
80 (m_arr[0] * m_arr[0]) +
81 (m_arr[1] * m_arr[1]) +
82 (m_arr[2] * m_arr[2]) +
83 (m_arr[3] * m_arr[3])
84 );
85
91 assert(!float_equality(magn, 0.0));

93 m_arr[0] /= magn;
94 m_arr[1] /= magn;
95 m_arr[2] /= magn;
96 m_arr[3] /= magn;

99[[nodiscard]] Vector<3> Quaternion::eigen_axis() const
100{
