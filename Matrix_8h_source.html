670template<std::size_t R, std::size_t C>
671std::ostream& operator<<(std::ostream& os, const Matrix<R,C>& mat)
672{
673 // figure out how many chars are required to print a number
674 auto num_str_chars = [](const double val){
675 std::ostringstream ss;
676 ss << val;
677 return ss.str().size();
678 };
679
680 // loop through array and find the longest number
681 std::size_t most_chars = 1;
682 for (std::size_t ii = 0; ii < mat.rows(); ii++)
683 {
684 for (std::size_t jj = 0; jj < mat.cols(); jj++)
685 {
686 const std::size_t chars = num_str_chars(mat(ii, jj));
687 most_chars = (chars > most_chars) ? chars : most_chars;
688 }
689 }
694 for (std::size_t ii = 0; ii < mat.rows(); ii++)
695 {
696 for (std::size_t jj = 0; jj < mat.cols(); jj++)
697 {
698 std::string prefix = (jj == 0) ? "\n" : ""; // put a newline or no char before the value
699 std::string suffix = (jj == mat.cols()-1) ? "" : ", "; // put a comma or no char after the value
700
701 os << prefix << std::setw(static_cast<int>(most_chars)) << std::left << mat(ii, jj) << suffix;
702 }
703 }
704
705 return os;
706}
