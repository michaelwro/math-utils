706  std::ostringstream ss;
707  ss << val;
708  return ss.str().size();
709  };
710 
711  // loop through array and find the longest number
712  std::size_t most_chars = 1;

714  {
715  for (std::size_t jj = 0; jj < mat.cols(); jj++)
716  {
717  const std::size_t chars = num_str_chars(mat(ii, jj));
718  most_chars = (chars > most_chars) ? chars : most_chars;
719  }
720  }
725  for (std::size_t ii = 0; ii < mat.rows(); ii++)
726  {
727  for (std::size_t jj = 0; jj < mat.cols(); jj++)
728  {
729  std::string prefix = (jj == 0) ? "\n" : ""; // put a newline or no char before the value

731 
732  os << prefix << std::setw(static_cast<int>(most_chars)) << std::left << mat(ii, jj) << suffix;
733  }
