687 std::ostream& operator<<(std::ostream& os, const Matrix<R,C>& mat)
688 {
689  // figure out how many chars are required to print a number
690  auto num_str_chars = [](const double val){
691  std::ostringstream ss;
692  ss << val;
693  return ss.str().size();
694  };
695 
696  // loop through array and find the longest number
697  std::size_t most_chars = 1;
698  for (std::size_t ii = 0; ii < mat.rows(); ii++)
699  {
700  for (std::size_t jj = 0; jj < mat.cols(); jj++)
701  {
702  const std::size_t chars = num_str_chars(mat(ii, jj));
703  most_chars = (chars > most_chars) ? chars : most_chars;
704  }
705  }
710  for (std::size_t ii = 0; ii < mat.rows(); ii++)
711  {
712  for (std::size_t jj = 0; jj < mat.cols(); jj++)
713  {
714  std::string prefix = (jj == 0) ? "\n" : ""; // put a newline or no char before the value
715  std::string suffix = (jj == mat.cols()-1) ? "" : ", "; // put a comma or no char after the value
716 
717  os << prefix << std::setw(static_cast<int>(most_chars)) << std::left << mat(ii, jj) << suffix;
718  }
719  }
720 
721  return os;
722 }
